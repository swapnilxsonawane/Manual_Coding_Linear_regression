while (err > ep and iter < 100):
    
    # Calculate Predicted value of Dependent Variable
    yh= w2*x2+ w1*x1 + w0;
    
    # Calculate DJ/DW: derivative of cost function wrt Weights.

    DJ0= yh-y;
    DJ1 =(yh-y)* x1;
    DJ2 =(yh-y)* x2;
   
    # This is normal gradient based descent. We are trying momentum based gradient descent.
    '''
    dw0=-alp* sum(DJ0) / m;
    dw1 = -alp* sum(DJ1) / m;
    dw2 = -alp* sum(DJ2) / m;
    '''
    # now I am introducing momentum based gradient descent. 25% less iterations but les accuracy fir weightages
    
    
    dw0=-(w0 * r +(alp* sum(DJ0) / m));
    dw1=-(w1 * r +(alp* sum(DJ1) / m));
    dw2=-(w2 * r +(alp* sum(DJ2) / m));
    
    
    # Updating new set of weights
    
    w0= w0 + dw0;
    w1 = w1 + dw1;
    w2 = w2 + dw2;

    # Updating Iteration 
    iter = iter + 1;
    itr.append(iter)
    
    #Updating Cost Function based upon new weight
    j.append (0.5/m *  sum ((y-w2*x2-w1*x1-w0)**2) );
    
   # j.append (0.5/m *  sum ((y-w1*x1-w0)**2) );
    
    err = abs(j[iter] - j[iter-1] );

    plt.subplot(211);
    plt.plot(itr,j,'o')
    plt.xlabel('iterations')
    plt.xlabel('cost')
    plt.show()
